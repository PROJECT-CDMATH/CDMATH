00024 /*
00025  * - Nom de la fonction : _MEDnomEntite
00026  * - Description : fournit le nom associe a un type d'entite MED
00027  * - Parametres :
00028  *     - nom_ent (OUT) : le nom de l'entite
00029  *     - type_ent (IN) : le type de l'entite
00030  * - Resultat : 0 en cas de succes, -1 sinon
00031  */ 
00032 med_err _MEDnomEntite(char *nom_ent,med_entite_maillage type_ent)
00033 {
00034    switch(type_ent)
00035      {
00036      case MED_NOEUD :
00037        strcpy(nom_ent,MED_NOM_NOE);
00038        break;
00039 
00040        /* MED_NOEUD_...ne devrait pas intervenir à ce niveau, 
00041           mais évite une modification sélective
00042           dans les API de gestion des champs */
00043      case MED_NOEUD_MAILLE :
00044        strcpy(nom_ent,MED_NOM_NOE);
00045        break;
00046 
00047      case MED_MAILLE :
00048        strcpy(nom_ent,MED_NOM_MAI);
00049        break;
00050 
00051      case MED_FACE :
00052        strcpy(nom_ent,MED_NOM_FAC);
00053        break;
00054 
00055      case MED_ARETE :
00056        strcpy(nom_ent,MED_NOM_ARE);
00057        break;
00058 
00059      default :
00060        return -1;
00061      }
00062    return 0;
00063 }
