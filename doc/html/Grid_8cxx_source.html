128 void
129 Grid::statistic(void) const
130 {
131  vector<int> rect_area(getNumberOfPatchs()) ;
132  int total_rect_area =0;
133  float min_nx = 1.E30 ;
134  float min_ny = 1.E30 ;
135  int flagged_area=0;
136  for (unsigned int i=0;i<getNumberOfPatchs();i++)
137  {
138  int nx = getPatch(i).getNx() ;
139  int ny = getPatch(i).getNy() ;
140  if ( float(nx) < min_nx )
141  min_nx = float(nx) ;
142  if ( float(ny) < min_ny )
143  min_ny = float(ny) ;
144  rect_area[i] = nx*ny ;
145  total_rect_area+=rect_area[i] ;
146  flagged_area+=getPatch(i).getNumberOfCellsFlags();
147  }
148  int min_edge = min(min_nx,min_ny) ;
149  int max_rect_area=*max_element(rect_area.begin(),rect_area.end()) ;
150  float mean_rect_area = float(total_rect_area) / float(rect_area.size()) ;
151  float ratio = 100.*flagged_area / total_rect_area ;
152  cout << "=====================================================" << endl;
153  cout << " Static " << endl;
154  cout << "=====================================================" << endl;
155 
156  cout << "- Total Number of flagged pts : " << flagged_area << endl;
157  cout << "- Total Area in patchs : " << total_rect_area << endl;
158  cout << "- flagged points/tot.patch.vol. : " << ratio << "%" << endl;
159  cout << "- Number of patchs : " << getNumberOfPatchs() << endl;
160  cout << "- Minimum box edge : " << min_edge << endl;
161  cout << "- Maximum box volume : " << max_rect_area << endl;
