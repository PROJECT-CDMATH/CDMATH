129 int
130 Mesh::getIndexFacePeriodic(int indexFace) const
131 {
132  if (!_faces[indexFace].isBorder())
133  return -1;
134  double minmax[4];
135  _mesh->getCoords()->getMinMaxPerComponent(minmax);
136  double xmin=minmax[0];
137  double xmax=minmax[1];
138  double ymin=minmax[2];
139  double ymax=minmax[3];
140  int nbFace=getNumberOfFaces();
141  double x=_faces[indexFace].x();
142  double y=_faces[indexFace].y();
143  double eps=1.E-10;
144  // pos=0 : bottom
145  // pos=1 : right
146  // pos=2 : top
147  // pos=3 : left
148  int pos=-1;
149  if (abs(y-ymin)<1.E-10)
150  pos=0;
151  if (abs(x-xmax)<1.E-10)
152  pos=1;
153  if (abs(y-ymax)<1.E-10)
154  pos=2;
155  if (abs(x-xmin)<1.E-10)
156  pos=3;
157  if (pos==-1)
158  cerr << "periodic face not found " << endl;
159 
160  int ifaceOk=-1;
161  for (int iface=0;iface<nbFace;iface++)
162  {
163  double xi=_faces[iface].x();
164  double yi=_faces[iface].y();
165  if (abs(y-yi)<eps && pos==1 && abs(xi-xmin)<eps)
166  {
167  ifaceOk=iface;
168  break;
169  }
170  if (abs(y-yi)<eps && pos==3 && abs(xi-xmax)<eps)
171  {
172  ifaceOk=iface;
173  break;
174  }
175  if (abs(x-xi)<eps && pos==0 && abs(yi-ymax)<eps)
176  {
177  ifaceOk=iface;
178  break;
179  }
180  if (abs(x-xi)<eps && pos==2 && abs(yi-ymin)<eps)
181  {
182  ifaceOk=iface;
183  break;
184  }
185  }
186  return ifaceOk;
187 }
