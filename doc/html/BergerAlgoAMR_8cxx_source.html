72 BergerAlgoAMR::computeGrid(const Mesh& mesh, const Field& fieldFlag, const int level)
73 {
74  unsigned int minCellDirection=getMinCellDirection()[level];
75  double effeciency=getEfficiency()[level];
76  double effeciencySnd=getEfficiencySnd()[level];
77  int maxCells=getMaxCells()[level] ;
78 
79  if (isVerbose())
80  {
81  cout << "/////////////////////////////////////////////" << endl;
82  cout << "// Parameters Level n° " << level << " //" << endl;
83  cout << "/////////////////////////////////////////////" << endl;
84  cout << "Efficiency threshold: " << 100*effeciency << "%"<< endl;
85  cout << "Efficiency Snd threshold: " << 100*effeciencySnd << "%"<< endl;
86  cout << "Min. box side length: " << minCellDirection << endl;
87  cout << "Max. box cells : " << maxCells << endl;
88  }
89 
90  /* Initialize and find the first box */
91  int coefRefinementX=getCoefRefinementX()[level];
92  int coefRefinementY=getCoefRefinementY()[level];
93 
94  Grid grid(level,coefRefinementX,coefRefinementY);
95 
96  Grid gridOut(level,coefRefinementX,coefRefinementY);
97 
98  Patch PInitial(mesh,fieldFlag);
99  PInitial.computeMesh(mesh);
100  PInitial.computeFieldFlag(fieldFlag);
101  grid.addPatch(PInitial);
102  unsigned int bigDims ;
103  int dissectDirection ;
104  int nX ;
105  int nY ;
106  unsigned int idPatch = 0 ;
107  int cut[3] ;
108  int nbActions=4 ;
109 
110  while ( idPatch < grid.getNumberOfPatchs() )
111  {
112  Patch patch=grid.getPatch(idPatch);
113  patch.computeMesh(mesh);
114  patch.computeFieldFlag(fieldFlag);
115  if (isVerbose())
116  patch.infoPatch();
117  nX= patch.getNx();
118  nY= patch.getNy();
119  if ( nY > nX )
120  {
121  bigDims = nY ;
122  dissectDirection = 1 ;
123  }
124  else
125  {
126  bigDims = nX ;
127  dissectDirection = 0 ;
128  }
129 
130  int cutFound= 0 ;
131  int cutPlace=-1 ;
132  int cutDim=-1 ;
133  int flagAction1=0 ;
134  int flagAction2=0 ;
135  int flagAction3=0 ;
136  int flagAction4=0 ;
137  for(int idAction=1; idAction<=nbActions; idAction++)
138  {
139  switch (idAction)
140  {
141  case 1 : /* Action 1 */
142  {
143  if (isVerbose())
144  cout << "- Action 1: check efficiency; if efficienct but box too big, dissect" << endl;
145  if ( patch.getEfficiency() >= effeciency )
146  {
147  flagAction1 = 0 ;
148  if (isVerbose())
149  cout << "- Box has the required efficiency" << endl;
150  if ( patch.getNumberOfCells() > maxCells)
151  {
152  patch.dissectBigPatch (mesh,
153  fieldFlag,
154  minCellDirection,
155  bigDims,
156  dissectDirection,
157  cut) ;
158  cutFound = cut[0] ;
159  cutPlace = cut[1] ;
160  cutDim = cut[2] ;
161  }
162  if (cutFound==0) flagAction1 = 1 ;
163  }
164  break ;
165  }
166  case 2 : /* Action 2 */
167  {
168  if (isVerbose())
169  cout << "- Action 2: looking for holes" << endl;
170  flagAction2=0;
171  patch.findHole(dissectDirection,
172  minCellDirection,
173  cut) ;
174  cutFound = cut[0] ;
175  cutPlace = cut[1] ;
176  cutDim = cut[2] ;
177  if (cutFound==1) flagAction2 = 1 ;
178  break ;
179  }
180  case 3 : /* Action 3 */
181  {
182  if (isVerbose())
183  cout << "- Action 3: looking for inflection points" << endl ;
184  flagAction3 = 0 ;
185  patch.findInflection(minCellDirection,
186  cut);
187  cutFound = cut[0] ;
188  cutPlace = cut[1] ;
189  cutDim = cut[2] ;
190  if (cutFound==1) flagAction3 = 1 ;
191  break ;
192  }
193  case 4 : /* Action 4 */
194  {
195  if (isVerbose())
196  cout << "- Action 4: no holes or inflections, dissect if not efficienct and not too small; otherwise, check if too big" << endl ;
197  flagAction4 = 0 ;
198  cut[2]=dissectDirection;
199  if ( patch.getEfficiency() <= effeciencySnd)
200  {
201  if (bigDims>=2*minCellDirection)
202  {
203  if (isVerbose())
204  {
205  cout << "Bisecting box because efficiency = " << patch.getEfficiency() << " < " << effeciencySnd
206  << " and side size = " << bigDims << " > minimum cells = " << minCellDirection << endl;
207  }
208  cut[0]=1;
209  cut[1]=int(bigDims/2)+patch.getIndexCorners()[dissectDirection]-1;
210  cutFound = cut[0] ;
211  cutPlace = cut[1] ;
212  cutDim = cut[2] ;
213  }
214  } else
215  {
216  if ( patch.getNumberOfCells() > maxCells)
217  {
218  patch.dissectBigPatch (mesh,
219  fieldFlag,
220  minCellDirection,
221  bigDims,
222  dissectDirection,
223  cut) ;
224  cutFound = cut[0] ;
225  cutPlace = cut[1] ;
226  cutDim = cut[2] ;
227  }
228  }
229  if (cutFound==1) flagAction4 = 1 ;
230  break ;
231  }
232  default:/* Action default */
233  {
234  cerr << "Unknown action! exiting" << endl ;
235  }
236  }
237  if ( flagAction1 ) break ;
238  if ( flagAction2 ) break ;
239  if ( flagAction3 ) break ;
240  if ( flagAction4 ) break ;
241  if ( cutFound ==1 ) break;
242  }
243  /* Patch found */
244  if ( cutFound )
245  {
246  if(isVerbose())
247  cout << "This box is dissected at cut.dim = " << cutDim << " cut.place = " << cutPlace << endl;
248  int rectLeft[4] ;
249  int rectRight[4] ;
250  patch.dissectPatch(mesh,
251  fieldFlag,
252  minCellDirection,
253  cut,
254  rectLeft,
255  rectRight);
256  Patch patchLeft(rectLeft);
257  patchLeft.computeMesh(mesh);
258  patchLeft.computeFieldFlag(fieldFlag);
259  grid.addPatch(patchLeft);
260  Patch patchRight(rectRight);
261  patchRight.computeMesh(mesh);
262  patchRight.computeFieldFlag(fieldFlag);
263  grid.addPatch(patchRight);
264  grid.erasePatch(idPatch);
265  }
266  else
267  {
268  patch.setIndex(idPatch);
269  patch.computeMesh(mesh,coefRefinementX,coefRefinementY);
270  gridOut.addPatch(patch);
271  cout << "- This patch " << idPatch+1 << " is accepted" << endl ;
272  idPatch++;
273  }
274  }
275  return gridOut ;
276 }
